\section*{Set Algebra}
\begin{align}
    \limsup E_n &= \bigcap_{n=1}^\infty \bigcup_{k=n}^\infty E_k\\ 
    \liminf E_n &= \bigcup_{n=1}^\infty \bigcap_{k=n}^\infty E_k\\ 
    \limsup f_n &= -\liminf \{-f_n\}\\ 
    \liminf f_n &= -\limsup\{-f_n\}
\end{align}

\tbf{Equivalence Relation:} $\sim$ is an equivalence relation on $X$ if 
\begin{enumerate}
    \item $x \sim x$ 
    \item $x \sim y \implies y \sim x$
    \item $x \sim y \land y \sim z \implies x \sim z$
\end{enumerate}

\tbf{Inverses:}
\begin{align}
    f^{-1}\left(\bigcup_{\alpha \in A} E_{\alpha}\right) &= \bigcup_{\alpha \in A} f^{-1}(E_{\alpha})\\ 
    f^{-1}\left(\bigcap_{\alpha \in A} E_{\alpha}\right) &= \bigcap_{\alpha \in A} f^{-1}(E_{\alpha})\\ 
    f^{-1}(E^c) &= (f^{-1}(E))^c
\end{align}

\tbf{Partial Ordering:}
\begin{enumerate}
    \item $xRy \land yRx \implies x = y$
    \item $xRy \land yRz \implies xRz$
    \item $xRx$ for all $x$
\end{enumerate}

\tbf{Total/Linear Ordering:}
\begin{enumerate}
    \item $R$ is a partial ordering 
    \item $xRy$ or $yRx$
\end{enumerate}

\tbf{Zorn's Lemma:} If $X$ is partially ordered and every totally ordered subset of $X$ is bounded, $X$ has a maximal element. 

\tbf{Well-Ordering:} 
\begin{enumerate}
    \item $X$ is linearly ordered 
    \item every nonempty subset of $X$ has a minimal element
\end{enumerate}

\tbf{Schroder-Bernstein:} $\exists f: X \inj Y, g: Y\inj X \implies \exists h: X \biject Y$

\tbf{Countability:}
\begin{itemize}
    \item $X, Y$ countable $\implies X \times Y$ countable 
    \item $X_n$ countable $\implies \bigcup_{n=1}^\infty X_n$ countable
\end{itemize}

\tbf{Topology of $\R$:} Every open set in $\R$ is a countable disjoint union of open intervals 

\tbf{Metric:} $\rho: X \times X \to [0, \infty)$ s.t. 
\begin{enumerate}
    \item $\rho(x, y) = 0 \iff x = y$
    \item $\rho(x, y) = \rho(y, x)$
    \item $\rho(x, z) \leq \rho(x, y) + \rho(y, z)$
\end{enumerate}

\section*{Point-set topology}
\begin{itemize}
    \item \tbf{Open Set:} $\forall x > 0$, $\exists \ep > 0$ s.t. $B(x, \ep) \sub U$
    \item \tbf{Closed Set:} $E^c$ open
    \item \tbf{Interior:} $\overset{\circ}{E} = \bigcup_{O \sub E} O$
    \item \tbf{Closure:} $\bar E = \bigcap_{E \sub K} K$
    \item \tbf{Dense:} $E \sub X$ dense in $X$ if $\bar E = X$
    \item \tbf{Nowhere Dense:} $(\bar E)^{\circ} = \emptyset$
    \item \tbf{Separable:} $\exists E \sub X$ countable, dense
    \item \tbf{Complete Metric Space:} All cauchy sequences have limit in $X$
    \item \tbf{Set diameter:} $\diam E = \sup\{\rho(x, y): x, y \in E\}$
    \item \tbf{Bounded set:} $\diam E < \infty$
    \item \tbf{Totally bounded set:} $E$ may be covered by finitely many balls
\end{itemize}

\tbf{Characterization of closed sets:} for $E \sub X$ and $x \in X$, the following are equivalent:
\begin{enumerate}
    \item $x \in \bar E$
    \item $E \cap B(x, \ep) \neq \emptyset$
    \item $\exists x_n \in E$ with $x_n \to x \in E$
\end{enumerate}

\tbf{Continuity:}
\begin{itemize}
    \item \tbf{Continuity at a point:} $\forall \ep > 0$, $\exists \delta > 0$ s.t. $\rho(x, y) < \delta_x \implies \rho(f(x), f(y)) < \ep$
    \item \tbf{Uniform Continuity:} $\exists \delta > 0$ $\forall \ep >0$ s.t. $\rho(x, y) < \delta \implies \rho(f(x), f(y)) < \ep$
    \item \tbf{Cauchy Sequence:} for $m n \geq N$, $\rho(x_n, x_m) < \ep$
\end{itemize}

\tbf{Continuity Lemmas:}
\begin{itemize}
    \item $f: X \to Y$ continuous iff $f^{-1}(U) \sub X$ open for all $U \sub Y$
    \item A closed subset of a complete metric space is complete 
    \item A complete subset of a metric space is closed 
\end{itemize}

\tbf{Characterization of Compactness:} the following are equivalent:
\begin{enumerate}
    \item $E$ complete and totally bounded 
    \item Every sequence in $E$ has a convergent subsequence with limit in $E$
    \item Every open cover of $E$ has a finite subcover
\end{enumerate}

\section*{Measure Theory}
\tbf{Algebra:} $\A \sub P(X)$ such that 
\begin{enumerate}
    \item $E \in \A \implies E^c \in \A$
    \item $\{E_k\}_1^N \in \A \implies \bigcup_1^N \E_k \in \A$
\end{enumerate}

\tbf{Sigma Algebra:} $\M \sub P(X)$ such that 
\begin{enumerate}
    \item $E \in \A \implies E^c \in \A$
    \item $\{E_k\}_1^{\infty} \in \A \implies \bigcup_1^{\infty} E_k \in \A$
\end{enumerate}

\tbf{Generated Sigma Algebra:} $M(E) = \bigcap_{E \sub \A} \A$ for algebras $\A$
 
\tbf{Borel Sigma Algebra:} $\sigma$-algebra generated by open sets 

\tbf{Lemma:} An algebra $\A$ is a $\sigma$-algebra if $\exists E_n \nearrow \in \A$ with $\bigcup_{j=1}^\infty E_j \in \A$

\tbf{Measure:} $\mu: X \to [0, \infty]$ such that 
\begin{enumerate}
    \item $\mu(\emptyset) = 0$
    \item $\mu\left(\bigcup_{n=1}^\infty E_n\right) = \sum_{n=1}^\infty E_n$ for $E_n$ disjoint 
\end{enumerate}

\tbf{Outer measure:} $\mu^*: X \to [0, \infty]$ with 
\begin{enumerate}
    \item $\mu^*(\emptyset) = 0$
    \item $\mu^*\left(\bigcup_{n=1}^\infty E_n\right) \leq \sum_{n=1}^\infty E_n$
\end{enumerate}

\begin{proof}
    \tbf{Properties of Measure:}
    \begin{enumerate}
        \item $E \sub F$, $\mu(E) \leq \mu(F)$
        \item $\mu\left(\bigcup_{j=1}^\infty E_j\right) \leq \sum_{j=1}^\infty \mu(E_j)$
        \item $E_j \nearrow$, 
        \[\mu\left(\bigcup_{j=1}^\infty E_j\right) = \lim_{j\to \infty} \mu(E_j)\]
        \item $E_j \searrow$ and $\mu(E_1) < \infty$, 
        \[\mu\left(\bigcap_{j=1}^\infty E_j\right) = \lim_{j \to \infty} \mu(E_j)\]
    \end{enumerate}
\end{proof}

\tbf{$\mu^*$-measurability:} $A$ measurable if $\forall E \sub P(X)$, 
\[\mu(E) = \mu(E \cap A) + \mu(E \cap A^c)\]

\tbf{Lebesgue Measure:} 
\[m(E) = \inf\left\{\sum \rho(E_j) \; | \; E \sub \bigcup E_j\right\} \]

\tbf{Properties of the Lebesgue Measure:}
\begin{itemize}
    \item $m(I) = m(I)$ for intervals $I$ on $\R$
    \item $m(E + a) = m(E)$
    \item $m(rE) = \abs{r}m(E)$
    \item $\exists O$ open such that $m(O) \geq m(E) \geq m(O) - \ep$
    \item $\exists K$ closed such that $m(K) \leq m(E) \leq m(K) + \ep$
\end{itemize}

\section*{Integration}
\tbf{Measurable Function:} $f: X \to \R$ measurable if $\{x: f(x) > \alpha\} \in \M$ or $\{f \geq \alpha\} \in \M$.

\tbf{Composition of Measurable Functions:}
\begin{itemize}
    \item $f, g$ measurable and $c \in \R$ $\implies$ $f + c$, $f + g$, $fg$, $cf$ measurable 
    \item $\{f_n\}$ measurable $\implies$ $\max_i f_i$, $\min_i f_i$, $\sup_n f_n$, $\inf_n f_n$, $\limsup_n f_n$, $\liminf_n f_n$ measurable
\end{itemize}

\tbf{Simple Functions:}
\begin{itemize}
    \item $\phi = \sum_{i=1}^{n} a_i \X_{E_i}$
    \item $\int \phi\; d\mu= \sum_{i=1}^{n} a_i \mu(E_i)$  
\end{itemize}

\tbf{Approximation by Simple Functions:}
\begin{itemize}
    \item $f \geq 0$ measurable implies $\exists \phi_n \nearrow f$, $\phi_n \to f$ pointwise on $X$, $\phi_n \to f$ uniformly on sets with $f$ bounded. 
    \item $f$ measurable implies $\exists 0 \leq \abs{\phi_1} \leq \abs{\phi_2} \leq \dots \leq \abs{f}$ with $\phi_n \to f$ pointwise on $X$ and uniformly on sets with $f$ bounded. 
\end{itemize}

\tbf{General Integral:} for $f \geq 0$
\[\int f \; d\mu = \sup_{\phi \leq f} \left\{\int \phi \; d\mu\right\}\]

\tbf{General Integral Properties:} for $f \geq 0$, 
\begin{itemize}
    \item $c \geq 0 \implies \int cf \; d\mu = c\int f \; d\mu$
    \item $\int f + g \; d\mu = \int f \; d\mu + \int g \; d\mu$
    \item $f \leq g \implies \int f \; d\mu \leq \int g \; d\mu$
\end{itemize}

\begin{proof}
    \tbf{Monotone Convergence Theorem:} if $f_n \geq 0$ and $f_n \leq f_{n+1}$, 
    \[\lim_{n \to \infty} \int f_n \; d\mu = \int \lim_{n \to \infty} f \; d\mu\]

    \tbf{Fatou's Lemma:} If $f_n \geq 0$, 
    \[\int \liminf f_n \; d\mu \leq \liminf \int f_n \; d\mu\]

    \tbf{Lebesgue Dominated Convergence:} If $0 \leq f_n \leq g$ for $g \in \L^1$, 
    \[\lim_{n \to \infty} \int f_n \; d\mu = \int \lim_{n \to \infty} f_n \; d\mu\]

\end{proof}

\tbf{Almost Everywhere Properties:}
\begin{enumerate}
    \item $\mu(E) = 0 \implies \int_E f = 0$ for $f \geq 0$
    \item $\int_X f = 0$ for $f \geq 0 \implies f = 0$ a.e. 
    \item $\int_X f < \infty \implies f< \infty$ a.e. 
\end{enumerate}

\tbf{$\L^1$ (integrable):} $\int\abs{f} \; d\mu < \infty$ 

\tbf{Properties of $\L^1$ functions:} $f, g \in \L^1$ and $a, b\in \R$:
\begin{enumerate}
    \item $af + bg \in \L^1$
    \item $\int af + bg \; d\mu = a\int f + b \int g$
    \item $\abs{\int f \; d\mu} \leq \int \abs{f}\; d\mu$
\end{enumerate}

\tbf{Approximation by simple functions:}$f \in \L^1 \implies \exists \phi$ simple s.t. $\int \abs{f - \phi} \; d\mu < \ep$

\tbf{Reduction to smooth functions:} For $\mu$ the Lebesgue-Stieltjes on $\R$, $\exists g \in C^{\infty}$ with compact support such that 
\[\int \abs{ f -g } \; d\mu < \ep\]

\tbf{Integrals with Parameter:} For $F(t) = \int f(x, t)\; d\mu$ integrable in $t$, 
\begin{itemize}
    \item If $\abs{f(x, t)} \leq g(x) \in \L^1$ and $f(x, t)$ is continuous in a.e. $t$, $F(t)$ is continuous 
    \item If $\exists g \in \L^1$ with $\abs{\frac{\partial f}{\partial t}\big\vert_{(x, t)}} \leq g(x)$, 
    \[F'(t) = \int \frac{\partial f}{\partial t}\bigg\vert_{(x, t)}\; d\mu\] 
\end{itemize}

\tbf{Lebesgue and Riemann:} $f$ bounded real, 
\begin{itemize}
    \item $f$ Riemann integrable $\implies f$ Lebesgue integrable with same value 
    \item $f$ Riemann integrable $\iff$ $f$ has countably many discontinuities
\end{itemize}

\begin{proof}
    \tbf{Modes of Convergence:}
    \begin{itemize}
        \item \tbf{Convergence in $\L^1$:} $\int \abs{f_n - f} \; d\mu \to 0$
        \item \tbf{Convergence in measure:} $\mu(\{x: \abs{f_n(x) - f(x)} \geq \ep\}) \to 0$
        \item \tbf{Chebysheb inequality:} 
        \[\int \abs{f_n - f} \; d\mu \geq \ep \mu(\{x: \abs{f_n(x) - f(x)} \geq \ep\})\] 
        \item \tbf{Cauchy in measure:} $\forall \ep, \delta > 0$, $\exists N$ s.t. $n, m \geq N \implies$ $\mu(\{x: \abs{f_n(x) - f_m(x)} \geq \ep\}) < \delta$
    \end{itemize}
    
    \tbf{Convergence theorems:}
    \begin{itemize}
        \item $f_n \to f \in \L^1 \implies f_n \overset{\mu}{\longrightarrow} f$
        \item $f_n$ Cauchy in measure $\implies f_{n_j} \to f$ a.e. and $f_n \overset{\mu}{\longrightarrow} f$
        \item $f_n \overset{\mu}{\longrightarrow} f \implies f_{n_j} \to f$ a.e.
        \item $f_n \to f \in \L^1 \implies f_{n_j} \to f$ a.e.
    \end{itemize}
    
    \tbf{Egorov:} $f_n \to f$ a.e. pointwise $\implies \exists E \st \mu(E) = 0$ and $f_n \to f$ uniformly on $E^c$
\end{proof}

\section*{Signed Measures}
\tbf{Signed Measure:} $\nu: \M \to \bar \R$ with 
\begin{enumerate}
    \item $\nu(\emptyset) = 0$
    \item $\nu$ assumes at most one of $\infty, -\infty$
    \item $\nu\left(\bigcup_{j=1}^\infty E_j\right) = \sum_{j=1}^\infty \nu(E_j)$ for $E_j$ disjoint 
    \item If $ \nu\left(\bigcup_{j=1}^\infty E_j\right) < \infty$, $\sum_{j=1}^\infty \nu(E_j) < \infty$
\end{enumerate}

\tbf{Properties of signed measures:}
\begin{itemize}
    \item If $E^j \nearrow$, $\nu\left(\bigcup_{j=1}^\infty E_j\right)  = \lim_{j \to \infty} \nu(E_j)$
    \item If $E_j \searrow$, $\nu\left(\bigcap_{j=1}^\infty E_j\right) = \lim_{j \to \infty} \nu(E_j)$
\end{itemize}

\tbf{Signed measure sets:} A set is 
\begin{itemize}
    \item \tbf{Positive} if $\forall F \sub E$, $\nu(F) \geq 0$
    \item \tbf{Negative} if $\forall F \sub E$, $\nu(F) \leq 0$
    \item \tbf{Null} if $\forall F \sub E$< $\nu(F) = 0$
\end{itemize}

\tbf{Lemma:}
\begin{itemize}
    \item Any measurable subset of a positive set is positive 
    \item The union of countably many positive sets is positive
\end{itemize}

\begin{proof}
    \tbf{Hahn Decomposition:} for $\nu$ signed measure, there exists a unique $P$ positive and $N$ negative with $X = P \cup N$ where $P \cap N = \emptyset$

    \tbf{Jordan Decomposition:} for $\nu$ signed measure, $\exists \nu^+, \nu^-$ unique such that $\nu = \nu^+ -\nu^-$ and $\nu^+ \perp \nu^-$
\end{proof}

\tbf{Total Variation:} $\abs{\nu} = \nu^+ + \nu^-$

\tbf{Mutual singularity:} $\mu \perp \nu \iff X = E \cup F \land \mu(E) = \nu(F) = 0$
\begin{itemize}
    \item $\nu \perp \mu \iff \abs{\nu} \perp \mu$
\end{itemize}

\tbf{Absolute continuity:} $\nu \ll \mu$ if $\mu(E) = 0 \implies \nu(E) = 0$
\begin{itemize}
    \item $\nu \ll \mu \iff \abs{\nu} \ll \mu \iff \nu^+ \ll \mu \land \nu^- \ll \mu$
    \item $\nu \ll \mu$ iff $\forall \ep > 0$, $\exists \delta > 0$ s.t. $\mu(E) < \delta \implies \abs{\nu(E)} < \ep$
    \item $f \; d\mu \ll \mu$ for $f \in \L^1$
\end{itemize}

\tbf{Lemma:} If $f \in \L^1_{\mu}$, then $\mu(E) < \delta_f \implies f\; d\mu(E) < \ep$

\tbf{Dichotomy of $\perp$ and $\ll$:} If $\mu, \nu$ positive, finite measures, either $\mu \perp \nu$ or exists a positive set $E$ WRT $\nu - \ep \mu$

\subsection*{Derivatives}
\tbf{Lebesgue-Radon-Nikodym Theorem:}$\nu$ sigma-finite and $\implies$ $\exists f\in \L^1_{\mu}$ unique a.e. such that 
\[\nu = \lambda + f\; d\mu \quad \lambda \perp \mu\]

\tbf{Radon-Nikodym Derivative:} $\nu \ll \mu \implies \exists f = \frac{d\nu}{d\mu} \implies \nu(E) = \int_E f\; d\mu$

\tbf{Chain Rule:} $\nu$ sigma finite, $\lambda, \mu$ positive, $\nu\ll \mu \ll \lambda$, 
\begin{enumerate}
    \item $g \in \L^1 \implies g \frac{d\nu}{d\mu} \in \L^1_{\mu}$ and 
    \[\int g\; d\nu = \int g \frac{d\nu}{d\mu} \; d\mu\] 
    \item $\nu \ll \lambda$ and 
    \[\frac{d\nu}{d\lambda} = \frac{d\nu}{d\mu} \cdot \frac{d\mu}{d\lambda}\]
\end{enumerate}

\tbf{Vitali Covering:} For $c < m\left(\bigcup_{B \in \Cc} B\right)$ open balls, $\exists \{B_j\}_{j=1}^k \in \Cc$ with $\sum_{j=1}^k m(B_j) > 3^{-n}c$

\tbf{Average Value:} For $f \in \L^1_{\text{loc}}$ (integrable on bounded measurable subsets), 
\[A_r f(x) = \frac{1}{m(B(r, x))} \int_{B(r, x)}f \; d\mu\]
is continuous in $r >0 $ and $x \in \R^n$ and
\begin{itemize}
    \item (Strong Derivative) $\lim_{r \to 0} A_r f(x) = f(x)$ a.e.
    \item (Weak Derivative) $\lim_{r \to 0} \norm{A_r f(x) - f(x)}_{\L^1} \to 0$ and $\exists r_n \to 0$ where $A_{r_n} f(x) \to f(x)$ a.e.
\end{itemize}

\tbf{Hardy-Littlewood Maximal Function:} for $f \in \L^1_{\text{loc}}$, 
\[H f(x) = \sup_{r >0} \frac{1}{m(B(r, x))} \int_{B(r, x)} \abs{f(y)}\; dy\]

\tbf{Maximal Theorem:} $\exists C$ such that for $f \in \L^1$, 
\[m(\{x: H f > \alpha\}) \leq \frac{C}{\alpha} \int_{\R^n} \abs{f(y)}\; dy\]

\tbf{$\L^1$-weak functions:} $g \in \L^1_W$ if $m\{x: \abs{g}>\alpha\} \leq \frac{C}{\alpha} \int \abs{g}\; d\mu$

\section*{Homework Results}

\tbf{Liminf/Limsup Fatou:}
For $\{E_j\} \in \M$, $\mu(\liminf E_j) \leq \liminf \mu(E_j)$. 

If $\mu\left(\bigcup_{j=1}^\infty E_j\right) < \infty$, $\mu(\limsup E_j) \geq \limsup \mu(E_j)$. 

\tbf{Outer measure:} For $\mu^*$ outer measure, 
\[\sum_{k=1}^\infty \mu^*(E_k) < \infty \implies \mu^*(\limsup E_k) =0\]

\tbf{Verifying Outer Measure:}
\begin{enumerate}
    \item $\mu^*(\emptyset) = 0$. Trivial (take singletons)
    \item Monotonicity. Trivial (choose cover of larger set, invoke inf)
    \item Subadditivity. Notice $\exists \{E_{jk}\}$ with 
    \[\sum \rho(E_{jk}) \leq \mu^*(A_j) + \frac{\ep}{2^j} \implies \bigcup_j A_j \sub \bigcup_j \bigcup_k E_{jk}\]
\end{enumerate}

\tbf{Hausdoff Outer Measure:}
\[\mathcal H_{\alpha}^{\ep}(A) = \inf\left\{\sum_k \diam(A_k)^{\alpha} \; \bigg\vert \; A \sub \bigcup A_k, \quad \diam A_k \leq \ep\right\}\]

\tbf{Hausdorff Measure:}
\[\mathcal H_{\alpha} = \lim_{\ep \to 0} \mathcal{H}_{\alpha}^{\ep}\]
satisfies $H_{\alpha}(A) < \infty \implies H_{\beta}(A) = 0$ for $\beta > \alpha$

\tbf{$\L^p$-spaces:}
\begin{itemize}
    \item \emph{Definition:} $f \in \L^p$ if $\int \abs{f}^p \; d\mu < \infty$ for $1 \leq p \leq \infty$
    \item \emph{Property:} For $\mu(X) < \infty$, $f \in \L^p \implies f \in \L^q$ for $1 \leq q \leq p$.
\end{itemize}

\tbf{Layercake Representation:} For $f \geq 0$ measurable, 
\[\int \abs f \; d\mu = \int_0^{\infty} \mu\{x: \abs{f(x)} > \lambda\} \; d\lambda\]

\tbf{Convergence results:}
\begin{itemize}
    \item  If $f_n \to f$ a.e. and $\lim_{n \to \infty} \int \abs{f_n} \; d\mu = \int \abs{f} \; d\mu$, 
    \[\lim_{n \to \infty} \int \abs{f_n - f} \; d\mu = 0\]

    \item If $f_n \overset{\mu}{\longrightarrow} f$, 
    \[\liminf_n \int \abs{f_n} \; d\mu \geq \int \abs{f} \; d\mu\]

    \item If $f_n \overset{\mu}{\longrightarrow} f$ and $\abs{f_n} \leq g \in \L^1$, $f_n \to f \in \L^1$
    
    \item $f_n \geq 0$, $f_n \to f$ a.e., $\int_X f_n \to \int_X f$, then for $E \sub X$, 
    \[\lim_{n \to \infty} \int_E f_n = \int_E f\]

    \item $f_n \overset{\mu}{\longrightarrow} f$ and $g_n \overset{\mu}{\longrightarrow} g$, 
    \[f_n + g_n \overset{\mu}{\longrightarrow} f + g\]

    \item $f_n \overset{\mu}{\longrightarrow} f$, $g_n \overset{\mu}{\longrightarrow} g$, and $\mu(X) < \infty$,
    \[f_n g_n \overset{\mu}{\longrightarrow} fg\]
\end{itemize}

\tbf{Riemann Lebesgue Theorem:} $f \in \L^1(\R)$, 
\[\lim_{n \to \infty} \int f(x) \cos(nx)\; dx = 0\] 

\tbf{N-Balls:} for $f \in \L^1(\R^d)$, 
\[H f(x) \geq \frac{C}{\abs{x}^d} \int \abs{f} \; d\mu\] 

\tbf{Continuity of Lebesgue Integral:} $f \in \L^1_{m}(\R)$. 
\begin{itemize}
    \item If $\abs{h} < \delta$, 
    \[\int_{\R} \abs{f(x + h) - f(x)} \; dx < \ep\]

    \item The following is continuous: 
    \[F(t) = \int_{-\infty}^x f(t)\; dt\] 
\end{itemize}

\tbf{Signed Measures:} If $\nu = \lambda - \mu$, $\lambda \geq \nu^+$ and $\mu \geq \nu^-$
